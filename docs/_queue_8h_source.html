115  pthread_mutex_lock( &myMutex ); //Lock the mutex so only one thread can append
116  while(myCount == mySize) {
117  pthread_cond_wait(&notFull, &myMutex); //The Queue is full, please wait until it is not full.

121  pthread_mutex_unlock( &myMutex );
122  } //Else...
123 }
