152 inline int saferand(int min, int max)
153 {
154  static std::mt19937* generator = nullptr;
155  #pragma omp threadprivate(generator)
156  std::uniform_int_distribution<int> dis(min, max);
157  if (!generator) {
158  unsigned seed = clock() + omp_get_thread_num();
159  generator = new std::mt19937(seed);
160  }
161  int val = dis(*generator);
162  return val;
163  // return (*generator)();
164 }
